class arrList : public List<T> {   //顺序表 ，向量
  private:                         //取值类型，取值空间
    T * aList;                     //连续内存
    int maxSize;                   //内存最大长度
    int curLen;                    //当前长度（已用的内存）
    int position;                  //当前处理位置
  public:
    arrList(const int size){       //创建新表，设置内存大小
      maxSize = size;
      aList = new T[maxSize];
      curLen = position =0;
    }
    ~arrList(){                    // 析构函数，消除该表实例
      delete [] aList;
    }
    void clear(){                  //清楚表中存储的内容，使其成为空表
      delete [] aList;
      curLen=position=0;
      aList = new T[maxSize];
    }
    
    int length();                   //返回当前实际长度
    bool append(const T value);     //在表尾添加元素
    bool insert(const int p, const T value);      //插入元素
    bool delete(const int p);                     //删除位置p 上的元素
    bool setValue(const int p, const T value);    // 设置元素值
    bool getValue(const int p, T&value);          //返回元素
    bool getPos(int &p, const T value);           //查找元素
};

//插入元素
template <class T> bool arrList<T> :: insert(const int p, const T value){
  int i;
  if(curLen >= maxSize)     //判断列表是否已满
  {
    cout<< "The list is overflow" << endl;
    return false;
  }
  if(p<0 || p>curLen){     //判断插入位置是否合适
    cout << "Insertion point is illegal" <<endl;
    return false;
  }
  for(i=curLen;i>p;i--){   
    aList[i] =aList[i-1];   //从表尾curLen-1开始往右移动到p
    aList[p]=value;       //位置p插入新元素 
  } 
  curLen++;              //表的实际长度+1
  return true;
};
