    bool Del_Min(sqList &L,ElemType &value)//删除顺序表L中的最小元素节点，并通过引用型参数value返回其值
//若删除成功，则返回true；否则返回false
{
if(L.length==0)
  return false;
 value=L.data[0];
 int pos=0;
 for(int i=1;i<L.length;i++)
   if(L.data[i]<value）
   {
     value=L.data[i];
     pos=i;//最后一个元素填补时，定位依据
    }
 L.data[pos]=L.data[L.length-1];//最后一个元素的下标为L.length-1
 L.length--;
 return true;
 }
